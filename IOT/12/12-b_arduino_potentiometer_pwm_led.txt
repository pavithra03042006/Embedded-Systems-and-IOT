
int potPin = A0;
int ledPin = 13;

void setup() {
  pinMode(ledPin, OUTPUT);
}

void loop() {

  int potValue = analogRead(potPin);

  int delayTime = map(potValue, 0, 1023, 100, 1000);

  digitalWrite(ledPin, HIGH);
  delay(delayTime);

  digitalWrite(ledPin, LOW);
  delay(delayTime);
}
