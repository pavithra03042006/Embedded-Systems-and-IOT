#include <reg51.h>

#define LCD P2
sbit RS = P3^0;
sbit EN = P3^1;

void delay()
{
    int i;
    for(i = 0; i < 20000; i++);
}

void lcd_cmd(unsigned char c)
{
    LCD = c;
    RS = 0;
    EN = 1;
    delay();
    EN = 0;
}

void lcd_data(unsigned char d)
{
    LCD = d;
    RS = 1;
    EN = 1;
    delay();
    EN = 0;
}

void main()
{
    unsigned char a = 25, b = 10, sum;

    sum = a + b;

    lcd_cmd(0x38);
    lcd_cmd(0x0C);
    lcd_cmd(0x01);

    lcd_data(sum + 48);

    while(1);
}
