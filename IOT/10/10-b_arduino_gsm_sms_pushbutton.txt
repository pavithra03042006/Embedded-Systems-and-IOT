10b(gsm code)

int buttonPin = 2;

void setup() {

  pinMode(buttonPin, INPUT);

  Serial.begin(9600);
}

void loop() {

  if(digitalRead(buttonPin) == HIGH) {

    Serial.println("AT+CMGF=1");
    delay(1000);

    Serial.println("AT+CMGS=\"+911234567890\"");
    delay(1000);

    Serial.println("Button Pressed");
    delay(1000);

    Serial.println("SMS Sent");

    delay(5000);
  }
}
