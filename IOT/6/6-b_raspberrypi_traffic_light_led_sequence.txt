from machine import Pin
import time

red = Pin(1, Pin.OUT) 
yellow = Pin(5, Pin.OUT) 
green = Pin(9, Pin.OUT) 

while True:
    green.value(1)
    yellow.value(0)
    red.value(0)
    time.sleep(3)

    green.value(0)
    yellow.value(1)
    red.value(0)
    time.sleep(0.5)

    green.value(0)
    yellow.value(0)
    red.value(1)
    time.sleep(2)

    green.value(0)
    yellow.value(1)
    red.value(0)
    time.sleep(0.5)
