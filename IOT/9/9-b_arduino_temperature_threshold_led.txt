
int sensor = A0;
int led = 13;

void setup() {
  pinMode(led, OUTPUT);
  Serial.begin(9600);
}

void loop() {

  float temp = (analogRead(sensor) * 5.0 / 1023 - 0.5) * 100;

  Serial.println(temp);

  if(temp > 30)
    digitalWrite(led, HIGH);
  else
    digitalWrite(led, LOW);

  delay(1000);
}
