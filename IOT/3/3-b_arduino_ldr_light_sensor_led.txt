int ldr = A0;
int led = 13;

int threshold = 500;

void setup()
{
    pinMode(led, OUTPUT);
    Serial.begin(9600);
}

void loop()
{
    int lightValue = analogRead(ldr);

    Serial.println(lightValue);

    if(lightValue < threshold)
    {
        digitalWrite(led, HIGH);
    }
    else
    {
        digitalWrite(led, LOW);
    }

    delay(500);
}
